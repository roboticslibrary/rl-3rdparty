ExternalProject_Add(
	libxslt
	DEPENDS libxml2
	#GIT_REPOSITORY https://gitlab.gnome.org/GNOME/libxslt.git
	#GIT_TAG v1.1.39
	URL https://gitlab.gnome.org/GNOME/libxslt/-/archive/v1.1.39/libxslt-v1.1.39.tar.gz
	#URL ${CMAKE_CURRENT_SOURCE_DIR}/libxslt-v1.1.39.tar.gz
	URL_MD5 8058ddbcae9bc4a6ab75327e748561de
	CMAKE_ARGS
		-DBUILD_SHARED_LIBS=${BUILD_SHARED_LIBS}
		-DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
		-DCMAKE_CFG_INTDIR=${CMAKE_CFG_INTDIR}
		-DCMAKE_FIND_PACKAGE_PREFER_CONFIG=ON
		-DCMAKE_INSTALL_PREFIX=${CMAKE_INSTALL_PREFIX}
		-DCMAKE_PREFIX_PATH=${CMAKE_PREFIX_PATH}
		-DCMAKE_TRY_COMPILE_CONFIGURATION=${CMAKE_TRY_COMPILE_CONFIGURATION}
		-DLIBXSLT_WITH_PYTHON=OFF
		-DLIBXSLT_WITH_TESTS=OFF
	INSTALL_DIR ${CMAKE_INSTALL_PREFIX}
)

ExternalProject_Add_Step(
	libxslt install-licenses
	DEPENDEES install
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/Copyright ${CMAKE_INSTALL_FULL_DATADIR}/licenses/libxslt/Copyright
)
