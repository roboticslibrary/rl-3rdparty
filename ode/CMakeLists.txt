ExternalProject_Add(
	ode
	#GIT_REPOSITORY https://bitbucket.org/odedevs/ode.git
	#GIT_TAG 0.16.3
	URL https://bitbucket.org/odedevs/ode/downloads/ode-0.16.3.tar.gz
	#URL ${CMAKE_CURRENT_SOURCE_DIR}/ode-0.16.3.tar.gz
	URL_MD5 b9f5efe527cb7b947a35caadf0bd67c7
	CMAKE_ARGS
		-DBUILD_SHARED_LIBS=${BUILD_SHARED_LIBS}
		-DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
		-DCMAKE_CFG_INTDIR=${CMAKE_CFG_INTDIR}
		-DCMAKE_INSTALL_PREFIX=${CMAKE_INSTALL_PREFIX}
		-DCMAKE_PREFIX_PATH=${CMAKE_PREFIX_PATH}
		-DCMAKE_TRY_COMPILE_CONFIGURATION=${CMAKE_TRY_COMPILE_CONFIGURATION}
		-DODE_WITH_DEMOS=OFF
		-DODE_WITH_TESTS=OFF
	INSTALL_DIR ${CMAKE_INSTALL_PREFIX}
)

ExternalProject_Add_Step(
	ode install-licenses
	DEPENDEES install
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/COPYING ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/COPYING
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/LICENSE.TXT ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/LICENSE.TXT
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/LICENSE-BSD.TXT ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/LICENSE-BSD.TXT
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/GIMPACT/GIMPACT-LICENSE-BSD.TXT ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/GIMPACT/GIMPACT-LICENSE-BSD.TXT
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/GIMPACT/GIMPACT-LICENSE-LGPL.TXT ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/GIMPACT/GIMPACT-LICENSE-LGPL.TXT
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/libccd/BSD-LICENSE ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/libccd/BSD-LICENSE
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/OPCODE/COPYING ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/OPCODE/COPYING
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/ou/LICENSE.TXT ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/ou/LICENSE.TXT
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/ou/LICENSE-BSD.TXT ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/ou/LICENSE-BSD.TXT
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/ou/LICENSE-LESSER.TXT ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/ou/LICENSE-LESSER.TXT
	COMMAND ${CMAKE_COMMAND} -E copy <SOURCE_DIR>/ou/LICENSE-ZLIB.TXT ${CMAKE_INSTALL_FULL_DATADIR}/licenses/ode/ou/LICENSE-ZLIB.TXT
)
